\newcommand{\articlequote}[3]{
	\begin{quotation}
		\begin{center}
			\textit{#1}
			% \ifthenelse{〈test expression〉}{〈true code〉}{〈false code〉}
			% \equal{〈string1〉}{〈string2〉}
			\ifthenelse{\equal{#2}{}}{}{\\ \medskip \textbf{#2}}
			\bigskip
		\end{center}
	#3
	\end{quotation}
}
